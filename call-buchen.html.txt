<script>
  /* ===== BOOKING SYSTEM ===== */
  const TIME_SLOTS = ['09:00','09:30','10:00','10:30','11:00','11:30','14:00','14:30','15:00','15:30','16:00','16:30','17:00','17:30'];
  const DAYS_DE = ['Sonntag','Montag','Dienstag','Mittwoch','Donnerstag','Freitag','Samstag'];
  const MONTHS_DE = ['Jan','Feb','Mär','Apr','Mai','Jun','Jul','Aug','Sep','Okt','Nov','Dez'];

  const bkDates = document.getElementById('bkDates');
  const bkTimes = document.getElementById('bkTimes');
  const bkPickerWrap = document.getElementById('bkPickerWrap');
  const bkForm = document.getElementById('bkForm');
  const bkSuccess = document.getElementById('bkSuccess');
  const bkSummaryValue = document.getElementById('bkSummaryValue');
  const bkSuccessMeta = document.getElementById('bkSuccessMeta');
  const bkHiddenTermin = document.getElementById('bkHiddenTermin');
  const bkBack = document.getElementById('bkBack');
  const bkFormEl = document.getElementById('bkFormEl');
  const bkSubmit = document.getElementById('bkSubmit');
  const stepEls = document.querySelectorAll('#bkSteps .step');

  let selectedDate = null;
  let selectedTime = null;

  function setStep(n) {
    stepEls.forEach(el => {
      const s = parseInt(el.dataset.step, 10);
      el.classList.remove('active', 'done');
      if (s < n) el.classList.add('done');
      else if (s === n) el.classList.add('active');
    });
  }

  function generateDates() {
    bkDates.innerHTML = '';
    const today = new Date();
    let count = 0;
    let offset = 1;
    while (count < 21) {
      const d = new Date(today);
      d.setDate(today.getDate() + offset);
      offset++;
      const dow = d.getDay();
      if (dow === 0 || dow === 6) continue;

      const btn = document.createElement('button');
      btn.type = 'button';
      btn.className = 'bk-date';
      btn.dataset.date = d.toISOString().split('T')[0];

      const dayName = DAYS_DE[dow];
      const dayNum = d.getDate();
      const month = MONTHS_DE[d.getMonth()];
      const isThisWeek = (d - today) / (1000*60*60*24) < 7;

      btn.innerHTML = `
        <span class="bk-date-day">${dayName}, ${dayNum}. ${month}</span>
        <span class="bk-date-meta">${isThisWeek ? 'diese Woche' : 'verfügbar'}</span>
      `;
      btn.addEventListener('click', () => selectDate(btn, d));
      bkDates.appendChild(btn);
      count++;
    }
  }

  function selectDate(btn, dateObj) {
    document.querySelectorAll('.bk-date').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    selectedDate = dateObj;
    selectedTime = null;
    renderTimes();
  }

  function renderTimes() {
    bkTimes.innerHTML = '';
    if (!selectedDate) {
      bkTimes.innerHTML = '<div class="bk-time-empty">Wähle zuerst ein Datum</div>';
      return;
    }
    TIME_SLOTS.forEach(time => {
      const btn = document.createElement('button');
      btn.type = 'button';
      btn.className = 'bk-time';
      btn.textContent = time;
      btn.addEventListener('click', () => selectTime(btn, time));
      bkTimes.appendChild(btn);
    });
  }

  function selectTime(btn, time) {
    document.querySelectorAll('.bk-time').forEach(b => b.classList.remove('selected'));
    btn.classList.add('selected');
    selectedTime = time;
    setTimeout(() => goToForm(), 350);
  }

  function goToForm() {
    const dateStr = `${DAYS_DE[selectedDate.getDay()]}, ${selectedDate.getDate()}. ${MONTHS_DE[selectedDate.getMonth()]} ${selectedDate.getFullYear()}`;
    const summary = `${dateStr} · ${selectedTime} Uhr`;
    bkSummaryValue.textContent = summary;
    bkHiddenTermin.value = summary;
    bkPickerWrap.classList.add('hidden');
    bkForm.classList.add('active');
    bkSuccess.classList.remove('active');
    setStep(2);
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  function backToPicker() {
    bkForm.classList.remove('active');
    bkPickerWrap.classList.remove('hidden');
    setStep(1);
  }

  bkBack.addEventListener('click', backToPicker);

  /* Form submit via Formspree (AJAX) */
  bkFormEl.addEventListener('submit', async e => {
    e.preventDefault();
    bkSubmit.disabled = true;
    bkSubmit.innerHTML = 'Wird gesendet...';

    const formData = new FormData(bkFormEl);
    const action = bkFormEl.getAttribute('action');

    try {
      const res = await fetch(action, {
        method: 'POST',
        body: formData,
        headers: { 'Accept': 'application/json' }
      });
      if (res.ok) {
        bkForm.classList.remove('active');
        bkSuccess.classList.add('active');
        bkSuccessMeta.textContent = bkSummaryValue.textContent;
        setStep(3);
        window.scrollTo({ top: 0, behavior: 'smooth' });
      } else {
        alert('Etwas ist schiefgelaufen. Bitte schreib uns direkt: hello@ilyd.io');
      }
    } catch (err) {
      alert('Etwas ist schiefgelaufen. Bitte schreib uns direkt: hello@ilyd.io');
    } finally {
      bkSubmit.disabled = false;
      bkSubmit.innerHTML = 'Termin bestätigen <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><path d="M5 12h14M13 6l6 6-6 6"/></svg>';
    }
  });

  /* Initialize */
  generateDates();
</script>
